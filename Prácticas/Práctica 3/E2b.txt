N_CLK EQU 10
N_F10 EQU 20
TIMER EQU 10H
CONT  EQU 10H
COMP  EQU 11H
PIC   EQU 20H
EOI   EQU 20H
IMR   EQU 21H
INT0  EQU 24H
INT1  EQU 25H
INT2  EQU 26H
INT3  EQU 27H
PIO   EQU 30H
PA    EQU 30H
PB    EQU 31H
CA    EQU 32H
CB    EQU 33H


ORG 1000H
MENSAJE DB "ORGANIZACION DE COMPUTADORAS"
FIN DB ?

ORG 2000H
MOV AL, 0FDH
OUT CA, AL
MOV AL, 0
OUT CB, AL

MOV BX, OFFSET MENSAJE
POOL: IN AL, PA
AND AL, 1
JNZ POOL

MOV AL, [BX]
OUT PB, AL

IN AL, PA
OR AL, 2
OUT PA, AL

IN AL, PA
AND AL, 0FDH
OUT PA, AL

INC BX
CMP BX, OFFSET FIN
JNZ POOL

LAZO: JMP LAZO

INT 0
END
