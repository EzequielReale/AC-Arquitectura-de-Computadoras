PA    EQU 30H
PB    EQU 31H
CA    EQU 32H
CB    EQU 33H

ORG 1000H
COMPARADOR DB 1,2,4,8,16,32,64,128

ORG 2000H
MOV AL, 0
OUT CB, AL
MOV BX, OFFSET COMPARADOR-1

CRECIENTE: INC BX
MOV AL, [BX]
OUT PB, AL
CMP AL, 128
JNZ CRECIENTE

DECRECIENTE: DEC BX
MOV AL, [BX]
OUT PB, AL
CMP AL, 1
JNZ DECRECIENTE

JMP CRECIENTE

INT 0
END
