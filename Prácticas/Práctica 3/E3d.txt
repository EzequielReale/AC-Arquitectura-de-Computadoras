EOI   EQU 20H
IMR   EQU 21H
INT2  EQU 26H
HAND_DATO EQU 40H
HAND_ESTADO EQU 41H
CONTADOR EQU 5

ORG 20
RUT_HAND DW IMPRIMIR

ORG 24
RUT_HAND_REVES DW IMPRIMIR_AL_REVES

ORG 1000H
BITS DB ?

ORG 3000H
IMPRIMIR: PUSH AX
MOV AL, [BX]
OUT HAND_DATO, AL
INC BX
DEC CL
JNZ FINAL

MOV AL, 0FFH
OUT IMR, AL

FINAL: MOV AL, 20H
OUT EOI, AL
POP AX
IRET

IMPRIMIR_AL_REVES: PUSH AX
DEC BX
MOV AL, [BX]
OUT HAND_DATO, AL
DEC CL
JNZ FIN 

MOV AL, 0FFH
OUT IMR, AL

FIN: MOV AL, 20H
OUT EOI, AL
POP AX
IRET

ORG 2000H
MOV CL, CONTADOR
MOV BX, OFFSET BITS
LEER: INT 6
INC BX
DEC CL
JNZ LEER

CLI 
MOV AL, 11111011B
OUT IMR, AL

MOV AL, 5
OUT INT2, AL

IN AL, HAND_ESTADO
OR AL, 10000000B
OUT HAND_ESTADO, AL

MOV BX, OFFSET BITS
MOV CL, CONTADOR
STI

LOOP: CMP CL, 0
JNZ LOOP

CLI
MOV AL, 11111011B
OUT IMR, AL

MOV AL, 6
OUT INT2, AL

MOV CL, CONTADOR
STI

LOOP_AL_REVES: CMP CL, 0
JNZ LOOP_AL_REVES

IN AL, HAND_ESTADO
AND AL, 011111111B
OUT HAND_ESTADO, AL

A: JMP A

INT 0
