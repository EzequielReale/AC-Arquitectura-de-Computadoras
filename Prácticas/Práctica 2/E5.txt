ORG 1000H
MSJ DB "INGRESE UN NUMERO:"
MSJ2 DB " CARACTER NO VALIDO"
FIN DB ?

ORG 1500H
NUM DB ?

ORG 3000H
ES_NUM:CMP CL, 30H
JS NO_ES
CMP CL, 40H
JNS NO_ES
RET

ORG 2000H
MOV BX, OFFSET MSJ
MOV AL, OFFSET MSJ2-OFFSET MSJ
INT 7
MOV BX, OFFSET NUM
INT 6
MOV CL, NUM
CALL ES_NUM

MOV AL, 1
INT 7
JMP FINAL

NO_ES:MOV BX, OFFSET MSJ2
MOV AL, OFFSET FIN-OFFSET MSJ2
INT 7

FINAL:INT 0
END
