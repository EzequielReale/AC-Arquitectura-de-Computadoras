EOI   EQU 20H
IMR   EQU 21H
INT0  EQU 24H

ORG 20
RUT_F10 DW CONTADOR

ORG 1000H
NUM DB ?

ORG 3000H
CONTADOR: PUSH AX
MOV AL, 1

ESCRIBO: INT 7
DEC NUM
CMP NUM, 2FH
JNZ ESCRIBO

DEC CL

MOV AL, 20H
OUT EOI, AL

POP AX
IRET

ORG 2000H
MOV BX, OFFSET NUM
INT 6

CLI
MOV AL, 11111110B
OUT IMR, AL

MOV AL, 5
OUT INT0, AL
STI

MOV CL, 1
LOOP: CMP CL, 0
JNZ LOOP

INT 0
END
